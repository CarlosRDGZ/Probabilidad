$$N = 500$$
$$n = 80$$
$$x = 20$$
$$p = \frac{20}{80}$$
$$p = 0.25$$

% b)
$$q = 1 - p$$
$$q = 1 - 0.25$$
$$q = 0.75$$

$$Sp^{2} = (\frac{N - n}{N})(\frac{p*q}{n})$$
$$Sp^{2} = (\frac{500 - 80}{500})(\frac{(0.25)(0.75)}{20})$$
$$Sp^{2} = 0.0019$$

$$Sp = \sqrt{Sp^{2}}$$
$$Sp = \sqrt{0.0019}$$
$$Sp = 0.0443$$

$$Z = 1.96$$

$$LS = p + Z*Sp$$
$$LS = 0.25 + (1.96)(0.0443)$$
$$LS = 0.3369$$
$$LI = p - Z*Sp$$
$$LI = 0.25 - (1.96)(0.0443)$$
$$LI = 0.1630$$

$$LI \leqslant p \leqslant LS$$
$$0.1630 \leqslant 0.25 \leqslant 0.3369$$

% c)
$$T = N*p$$
$$T = (500)(0.25)$$
$$T = 125$$
